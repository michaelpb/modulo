Example: ToDo App

<component name="Todo">
<template>
    <ol>
        <li><h1>To Do!</h1></li>
        {% for item in state.list %}
            <li>{{ item }}</li>
        {% endfor %}
        <li>
            <input [state.bind] name="text" />
            <button @click:=script.addItem>
                Add "{{ state.text }}"</button>
        </li>
    </ol>
</template>
<state
    list:='["Milk", "Bread", "Candy"]'
    text="Beer"
></state>
<script>
    function addItem() {
        state.list.push(state.text);
        state.text = '';
    }
</script>
</component>
